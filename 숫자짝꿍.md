def solution(X, Y):
    temp_X = []
    temp_Y = []
    
    answer = []
    
    for h in X:
        temp_X.append(h)
    temp_X.sort()
        
    for g in Y:
        temp_Y.append(g)
    temp_Y.sort()
    
    for g in temp_X:
        if g in temp_Y:
            answer.append(g)
            temp_Y.remove(g)
        else:
            continue
    return method(answer)
    
def method(list):
    if list.count('0') == len(list):
        return "0"
    elif len(list) == 0:
        return "-1"
    else:
        str = sorted(list,reverse = True)
        return "".join(str)
