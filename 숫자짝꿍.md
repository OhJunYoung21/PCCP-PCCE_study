def solution(X, Y):

    Y_1 = Counter(Y)

    for key,value in X_1.items():

            X_1[key] -= 1
            Y_1[key] -= 1
            answer += key 
    
    
    if not answer:
        return "-1"

        return "0"
    else:
        return answer
