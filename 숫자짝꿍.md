# 숫자짝꿍

~~~python3
def solution(X, Y):
    
    #리스트로 변환
    c1 = list(X)
    c2 = list(Y)
    
    temp = []
    
    for j in c1:
        check = j
        for h in c2:
            if h == check:
                temp.append(h)
                c2.remove(h)
                
            else:
                continue
    
    if len(temp) == 0:
        return "-1"
    else:
        if temp.count('0') == len(temp):
            return "0"
        else:
            answer = ''
            c3 = []
            for k in temp:
                c3.append(int(k))
            c3.sort(reverse=True)
            for j in c3:
                answer += str(j)
            return answer
~~~
