# 구현문제

좌표 내에서 문제가 원하는 상황을 구현하는 문제이다.

~~~python
def solution(park, routes):
    
    coordinate = [0,0]
    
    #S의 좌표를 구합니다.
    
    for h in range(len(park)):
        
        cnt = 0
        
        for g in park[h]:
            if g == "S":
                coordinate = [h,cnt]
                break
            else:
                cnt += 1 
                
    #coordinate에는 현재 S의 좌표가 담겨있다.
    
    coordinate_final = [coordinate[0],coordinate[1]]
    
    for h in routes:
        temp = h.split(" ")
        
        move = int(temp[1])
        #동쪽 이동인경우,
        if temp[0] == "E":
            for h in range(move):
                #일단 움직이고, 해당 결과가 옳게 움직였는지 확인한다.
                coordinate_final[1] += 1
                #오른쪽으로 너무 많이 움직여서 범위를 벗어난 경우, 변화 초기화
                if coordinate_final[1] >= len(park[0]):
                    coordinate_final = coordinate
                elif park[coordinate_final[0]][coordinate_final[1]] == "X":
                    coordinate_final = coordinate
                else:
                    continue
        elif temp[0] == "S":
            for h in range(move):
                #일단 움직이고, 해당 결과가 옳게 움직였는지 확인한다.
                coordinate_final[0] += 1
                #오른쪽으로 너무 많이 움직여서 범위를 벗어난 경우, 변화 초기화
                if coordinate_final[0] >= len(park):
                    coordinate_final = coordinate
                elif park[coordinate_final[0]][coordinate_final[1]] == "X":
                    coordinate_final = coordinate
                else:
                    continue
        elif temp[0] == "W":
            for h in range(move):
                #일단 움직이고, 해당 결과가 옳게 움직였는지 확인한다.
                coordinate_final[1] -= 1
                #오른쪽으로 너무 많이 움직여서 범위를 벗어난 경우, 변화 초기화
                if coordinate_final[1] < 0:
                    coordinate_final = coordinate
                elif park[coordinate_final[0]][coordinate_final[1]] == "X":
                    coordinate_final = coordinate
                else:
                    continue
        elif temp[0] == "N":
            for h in range(move):
                #일단 움직이고, 해당 결과가 옳게 움직였는지 확인한다.
                coordinate_final[0] -= 1
                #오른쪽으로 너무 많이 움직여서 범위를 벗어난 경우, 변화 초기화
                if coordinate_final[0] < 0:
                    coordinate_final = coordinate
                elif park[coordinate_final[0]][coordinate_final[1]] == "X":
                    coordinate_final = coordinate
                else:
                    continue
    return coordinate_final
~~~
