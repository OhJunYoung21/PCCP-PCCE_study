def solution(players, callings):
    
    temp_dict = {player:int(idx) for idx,player in enumerate(players)}
    
    
    for h in callings:
        
        rank = temp_dict[h]
        
        temp_dict[h] -= 1
        
        temp_dict[players[rank-1]] += 1
        
        players[rank-1], players[rank] = h,players[rank-1]
    return players
